# Atividades 3° Trismestre

 *Chave Estrangeira e Manipulação de dados

CREATE TABLE alunos(
  nome int NOT NULL,
  idade int NOT NULL,
  matricula int NOT NULL,
  PRIMARY KEY (matricula)
  )
  
  CREATE TABLE professores(
  nome int NOT NULL,
  idade int NOT NULL,
  matricula int NOT NULL,
  PRIMARY KEY (matricula)
  )
  
  CREATE TABLE pais(
    nome int NOT NULL,
    cpf int NOT NULL,
    dinheiro int NOT NULL,
    filho int NOT NULL,
    PRIMARY KEY (cpf)
    )
    FOREIGN key (filho)
    REFERENCES alunos(matricula)
    
    
    CREATE TABLE salaDeAula(
      numero int NOT NULL,
      quadro int not NULL,
      mesas int not NULL,
      PRIMARY KEY (numero)
